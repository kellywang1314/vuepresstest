import { KitsWrapper } from '../__component__/kitsWrapper'

export default class Example extends Page {
    render() {
        return (
            <ViewPort>
                <KitsWrapper forward={this.push.bind(this)}>
                    <YourPage/>
                </KitsWrapper>
            </ViewPort>
        )
    }
}
